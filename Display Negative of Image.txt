I = imread('bhupen.jpg');
J = rgb2gray(I);
K = im2bw(I);
I2 = ((2 ^ 8) - 1) - I;
J2 = ((2 ^ 8) - 1) - J;
K2 = ((2 ^ 1) - 1) - K;
subplot(3,2,1);
imshow(I);
title('Color Image');
subplot(3,2,2);
imshow(J);
title('Grayscale Image');
subplot(3,2,3);
imshow(K);
title('Binary Image');
subplot(3,2,4);
imshow(I2);
title('Negative of Color Image');
subplot(3,2,5);
imshow(J2);
title('Negative of Grayscale Image');
subplot(3,2,6);
imshow(K2);
title('Negative of Binary Image');
