j = imread('bhupen.jpg');
i = rgb2gray(j);
avgFilter = fspecial('average');
diskFilter = fspecial('disk');
gaussianFilter = fspecial('gaussian');
laplacianFilter = fspecial('laplacian');
logFilter = fspecial('log');
motionFilter = fspecial('motion');
prewittFilter = fspecial('prewitt');
sobelFilter = fspecial('sobel');
avgImg = imfilter(i, avgFilter);
diskImg = imfilter(i, diskFilter);
gaussianImg = imfilter(i, gaussianFilter);
laplacianImg = imfilter(i, laplacianFilter);
logImg = imfilter(i, logFilter);
motionImg = imfilter(i, motionFilter);
prewittImg = imfilter(i, prewittFilter);
sobelImg = imfilter(i, sobelFilter);
subplot(3,3,1);
imshow(i);
title("Original Image");
subplot(3,3,2);
imshow(avgImg);
title("Average Filter");
subplot(3,3,3);
imshow(diskImg);
title("disk Filter");
subplot(3,3,4);
imshow(gaussianImg);
title("Gaussian Filter");
subplot(3,3,5);
imshow(laplacianImg);
title("Laplacian Filter");
subplot(3,3,6);
imshow(logImg);
title("Log Filter");
subplot(3,3,7);
imshow(motionImg);
title("Motion Filter");
subplot(3,3,8);
imshow(prewittImg);
title("Prewitt Filter");
subplot(3,3,9);
imshow(sobelImg);
title("Sobel Filter");
