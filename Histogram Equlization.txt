A = imread('bhupen.jpg');
I = rgb2gray(A);
J = histeq(I);
subplot(2,2,1);
imshow(I);
title('Original Image');
subplot(2,2,2);
imhist(I,64);
title('Original Histogram');
subplot(2,2,3);
imshow(J);
title('Histogram Equalized Image');
subplot(2,2,4);
imhist(J,64);
title('Equalized Histogram');
