i = imread('bhupen.jpg');
subplot(2,2,1);
imshow(i);
title('Original Image');
r = i(:,:,1);
g = i(:,:,2);
b = i(:,:,3);
x = zeros(size(i,1),size(i,2));
red = cat(3, r, x, x);
subplot(2,2,2);
imshow(red);
title('Red Channel');
green = cat(3, x, g, x);
subplot(2,2,3);
imshow(green);
title('Green Channel');
blue = cat(3, x, x, b);
subplot(2,2,4);
imshow(blue);
title('Blue Channel');
